Future<List<DeviceModel>> getDevices() async {
  final snapshot = await _db.collection(_Collection.devices).get();

  return snapshot.docs
      .map((doc) => DeviceModel.fromJson(doc.data()).copyWith(id: doc.id))
      .toList();
}
